The indoor temperature dynamics are modeled by \eqref{Q}.
\begin{subequations}\label{Q}
\begin{align}
&\frac{\mathrm{d}T^{\mathrm{in}}}{\mathrm{d}t} = \frac{Q^{\mathrm{heat}} - Q^{\mathrm{vent}} - Q^{\mathrm{cool}} + Q^{\mathrm{solar}} - Q^{\mathrm{i-o}} + Q^{\mathrm{LED}} - Q^{\mathrm{tran}}}{\omega^{\mathrm{T}} \cdot V^{\mathrm{gh}} \cdot c^{\mathrm{vh}}}                                                                             
                                                                                       \label{Q:Tdyn}\\
& Q^{\mathrm{heat}} = \bar{Q}^{\mathrm{heat}} \cdot U^{\mathrm{heat}}                  \label{Q:heat}\\
& Q^{\mathrm{vent}} = c^{\mathrm{vh}} \cdot T^{\mathrm{i-o}} \cdot V^{\mathrm{vent}}   \label{Q:Qvent}\\
& Q^{\mathrm{cool}} = \bar{Q}^{\mathrm{pad}} \cdot U^{\mathrm{pad}} + q^{\mathrm{evap}} \cdot \eta^{\mathrm{evap}} \cdot \bar{F}^{\mathrm{hum}} \cdot U^{\mathrm{hum}}
                                                                                       \label{Q:cool}\\
& Q^{\mathrm{solar}} = A^{\mathrm{floor}} \cdot R^{\mathrm{cano-solar}}                \label{Q:Qsol}\\
& Q^{\mathrm{i-o}} = \left[
\upsilon^{\mathrm{roof}} \cdot A^{\mathrm{roof}} \cdot
\left(1-\eta^{\mathrm{warm}} \cdot U^{\mathrm{warm}}\right)
+ \upsilon^{\mathrm{wall}} \cdot  A^{\mathrm{wall}}
\right] \cdot T^{\mathrm{i-o}}                                                             \label{Q:QDelta}\\
& Q^{\mathrm{LED}}  = \bar{P}^{\mathrm{LED}} \cdot U^{\mathrm{LED}}                        \label{Q:QLED}\\
& Q^{\mathrm{tran}} = q^{\mathrm{evap}} \cdot g^{\mathrm{tran}} \cdot \left(H^{\mathrm{cano}} - H^{\mathrm{in}}\right) \cdot A^{\mathrm{floor}}                                                                    
                                                                                           \label{Q:Qtrans}
\end{align}
\end{subequations}

The indoor humidity dynamics are modeled by \eqref{H}.
\begin{subequations}\label{H}
\begin{align}
& \frac{\mathrm{d}H^{\mathrm{in}}}{\mathrm{d}t}
=\frac{F^{\mathrm{cool}} - F^{\mathrm{deh}} - F^{\mathrm{vent}} + F^{\mathrm{tran}}-F^{\mathrm{vc}}}{\omega^{\mathrm{H}} \cdot V^{\mathrm{gh}}}                                                                          
                                                                                    \label{H:Hdyn}\\
& F^{\mathrm{cool}}  = Q^{\mathrm{cool}} / q^{\mathrm{evap}}                        \label{H:Hcool}\\
& F^{\mathrm{deh}} = \bar{F}^{\mathrm{deh}}\cdot U^{\mathrm{deh}}             \label{H:Hdeh}\\
& F^{\mathrm{vent}}  = (H^{\mathrm{in}} - H^{\mathrm{out}}) \cdot V^{\mathrm{vent}} \label{H:Hvent}\\
& F^{\mathrm{tran}}  = Q^{\mathrm{tran}} / q^{\mathrm{evap}}                        \label{H:Htrans}\\
& F^{\mathrm{vc}}    =  g^{\mathrm{vc}}\cdot \left[\zeta \cdot \exp \left(\delta^{\mathrm{vc}} \cdot T^{\mathrm{in}} \right)\cdot T^{\mathrm{i-o}} - (H^{\mathrm{in,sat}} - H^{\mathrm{in}}) \right] \cdot A^{\mathrm{floor}}
                                                                                   \label{H:Hvc}
\end{align}
\end{subequations}

The indoor CO$_{2}$ concentration dynamics are modeled by \eqref{C}.
\begin{subequations}\label{C}
\begin{align}
& \frac{\mathrm{d}C^{\mathrm{in}}}{\mathrm{d}t} = \frac{D^{\mathrm{dos}} - D^{\mathrm{vent}} - D^{\mathrm{ass}}}{\omega^{\mathrm{C}} \cdot V^{\mathrm{gh}}} 
                                                                                              \label{C:Cdyn}\\
& D^{\mathrm{dos}}  = \bar{D}^{\mathrm{dos}} \cdot U^{\mathrm{dos}}                           \label{C:Cdos}\\
& D^{\mathrm{vent}} = (C^{\mathrm{in}} - C^{\mathrm{out}}) \cdot V^{\mathrm{vent}}            \label{C:Cvent}\\
& D^{\mathrm{ass}}  = \bar{D}^{\mathrm{ass}} \cdot \frac{C^{\mathrm{in}}}{C^{\mathrm{in}} + \hat{C}^{\mathrm{in}}}\cdot \left[1 - \exp\left(- \rho \cdot X^{\mathrm{cano}}\right)\right] \cdot A^{\mathrm{floor}}     \label{C:Cass}
\end{align}
\end{subequations}

The daily light integral is modeled by \eqref{DLI}:
\begin{equation}\label{DLI}
\frac{\mathrm{d}L}{\mathrm{d}t} =
\frac{X^{\mathrm{cano}}}{10^{6}}
\end{equation}

Note that the transpiration model in \eqref{Q:Qtrans} and \eqref{H:Htrans} is adopted from \cite{PJM_Minimal_heating_and_cooling_in_a_modern_rose_greenhouse, Stanghellini_A_Model_of_Humidity_1995}. The condensation vapor humidity model in \eqref{H:Hvc} is adopted from \cite{Stanghellini_A_Model_of_Humidity_1995, Bontsema_Jan_Online_estimation_of_the_transpiration}. The CO$_2$ assimilation model in \eqref{C:Cass} is adopted from \cite{PJM_Optimal_control_of_greenhouse, Stanghellini_ISHS_2008, Stanghellini_ISHS_2012}.

Hereafter, we introduce the intermediate variables. 

\begin{align}
& T^{\mathrm{i-o}} = T^{\mathrm{in}} - T^{\mathrm{out}}\\
& V^{\mathrm{vent}} =
\bar{V}^{\mathrm{fan}} \cdot U^{\mathrm{fan}}+
\bar{V}^{\mathrm{nat}} \cdot U^{\mathrm{nat}} +
\frac{\lambda^{\mathrm{leak}} \cdot V^{\mathrm{gh}}}{3600}
\end{align}

We follow references \cite{PJM_Minimal_heating_and_cooling_in_a_modern_rose_greenhouse, Stanghellini_A_Model_of_Humidity_1995, Bontsema_Jan_Online_estimation_of_the_transpiration, Goudriaan_A_Mathematical_Function_for_Crop_Growth} to calculate $g^{\mathrm{tran}}$:
\begin{align}
& g^{\mathrm{tran}} = \frac{2 \cdot I}{\left[1 + \eta^{\mathrm{tran}} \cdot \exp \left(\delta^{\mathrm{tran}} \cdot T^{\mathrm{in}}\right)\right] \cdot r^{\mathrm{b}} + r^{\mathrm{s}}}\label{g_tran}\\
& r^{\mathrm{s}} = \left[\underline{r}^{\mathrm{s}} + \bar{r}^{\mathrm{s}} \cdot \exp \left(-\frac{\tau \cdot R^{\mathrm{cano-abs}}}{I} \right)\right] \cdot \left[1 + \delta^{\mathrm{sr}} \cdot(T^{\mathrm{in}} - T^{\mathrm{sr}})^{2}\right]\\
& R^{\mathrm{cano-abs}} = \eta^{\mathrm{short}} \cdot \left[\frac{\exp \left(\eta^{\mathrm{ext}} \cdot I \right) - 1}{\exp \left(\eta^{\mathrm{ext}} \cdot I \right)}\right] \cdot R^{\mathrm{cano-glo}}\\
& R^{\mathrm{cano-glo}} = R^{\mathrm{cano-solar}} + R^{\mathrm{cano-LED}}\\
& R^{\mathrm{cano-solar}} = \eta^{\mathrm{cover}} \cdot \left(1-\eta^{\mathrm{shad}} \cdot U^{\mathrm{shad}}\right) \cdot R^{\mathrm{out}}\\
& R^{\mathrm{cano-LED}} = \frac{ \eta^{\mathrm{LED-r}} \cdot \eta^{\mathrm{LED-cano}} \cdot \bar{P}^{\mathrm{LED}} \cdot U^{\mathrm{LED}}}{A^{\mathrm{floor}}}               
\end{align}

We follow references \cite{PJM_Minimal_heating_and_cooling_in_a_modern_rose_greenhouse, Stanghellini_A_Model_of_Humidity_1995, Bontsema_Jan_Online_estimation_of_the_transpiration} to calculate $H^{\mathrm{cano}}$:
\begin{align}
&H^{\mathrm{cano}} = H^{\mathrm{in,sat}}+ \chi \cdot \frac{r^{\mathrm{b}} \cdot R^{\mathrm{cano-abs}}}{2 \cdot I \cdot q^{\mathrm{evap}}}\\
&H^{\mathrm{in,sat}}  = \tilde{H} \cdot \exp \left(\delta^{\mathrm{sat}} \cdot T^{\mathrm{in}}\right)
\end{align}

We follow references \cite{Stanghellini_A_Model_of_Humidity_1995, Bontsema_Jan_Online_estimation_of_the_transpiration}  to calculate $g^{\mathrm{vc}}$:
\begin{align}
&g^{\mathrm{vc}} = \nu^{\mathrm{surf}} \cdot
\left[
\frac{\left(T^{\mathrm{in}} - T^{\mathrm{cover}} \right) + \sqrt{\left(T^{\mathrm{in}} - T^{\mathrm{cover}}\right)^{2} + \sigma^{2}}}
{2} \right]^{1/3}\\
&T^{\mathrm{cover}} = \frac{2 \cdot T^{\mathrm{out}} + T^{\mathrm{in}}}{3}
\end{align}

We follow \cite{Stanghellini_ISHS_2012} to calculate the photon flux density of photosynthetically active radiation, known as photosynthetic photon flux density (PPFD):
\begin{equation}
X^{\mathrm{cano}}=\phi^{\mathrm{solar}} \cdot R^{\mathrm{cano-solar}} + \phi^{\mathrm{LED}} \cdot R^{\mathrm{cano-LED}}
\end{equation}

Next, we define more variables and parameters to build the MPC framework.

The state vector is defined as
\begin{equation}
\boldsymbol{x} = 
\left[
T^{\mathrm{in}},
H^{\mathrm{in}},
C^{\mathrm{in}},
L^{\mathrm{DLI}}
\right]^{\top}
\end{equation}

By default, the initial state vector is set as
\begin{equation}
\boldsymbol{x}^{\mathrm{ini}} = \left[
19   \,\si{\celsius},
7.41 \,\mathrm{g/m^{3}},
0.92 \,\mathrm{g/m^{3}},
0    \,\mathrm{\mathrm{mol/m^{2}}}
\right]^{\top}
\end{equation}

The control input vector is defined as
\begin{equation}
\boldsymbol{u} =
\left[
U^{\mathrm{heat}},
U^{\mathrm{fan}},
U^{\mathrm{nat}},
U^{\mathrm{pad}},
U^{\mathrm{dos}},
U^{\mathrm{LED}},
U^{\mathrm{hum}},
U^{\mathrm{deh}},
U^{\mathrm{shad}},
U^{\mathrm{warm}}
\right]^{\top}
\end{equation}

The outdoor disturbance vector is defined as
\begin{equation}
\boldsymbol{d} = 
\left[
T^{\mathrm{out}},
H^{\mathrm{out}},
C^{\mathrm{out}},
R^{\mathrm{out}}
\right]^{\top}
\end{equation}

The nonlinear continuous-time dynamics can be compactly written as
\begin{equation}\label{state_space_con}
\dot{\boldsymbol{x}}(t)=f\!\left(\boldsymbol{x}(t),\boldsymbol{u}(t),\boldsymbol{d}(t)\right),
\end{equation}
where $f(\cdot)$ is defined by the temperature, humidity, CO$_2$, and DLI dynamics in \eqref{Q}, \eqref{H}, \eqref{C}, and \eqref{DLI}.

Within the MPC framework, the continuous-time state-space model can be discretized using integration with time step $\Delta t$, yielding \eqref{statespace_d}:
\begin{equation}\label{statespace_d}
\boldsymbol{x}_{k+1} = \boldsymbol{x}_{k} + f(\boldsymbol{x}_{k},\boldsymbol{u}_{k},\boldsymbol{d}_{k}) \cdot \Delta t 
\end{equation}

Cost of resource consumption of the greenhouse during step $k$:
\begin{align}
& E^{\mathrm{heat}}_{k} =\alpha^{\mathrm{heat}}\cdot Q^{\mathrm{heat}}_{k} \cdot \frac{\Delta t}{3.6 \times 10^{6}}\\
& E^{\mathrm{fan}}_{k}  =\alpha^{\mathrm{fan}}\cdot S^{\mathrm{fan}}\cdot\bar{V}^{\mathrm{fan}}\cdot U^{\mathrm{fan}}_{k}\cdot\frac{\Delta t}{3.6\times 10^{6}}\\
& E^{\mathrm{LED}}_{k}  =\alpha^{\mathrm{LED}} \cdot \bar{P}^{\mathrm{LED}}   \cdot U^{\mathrm{LED}}_{k} \cdot \frac{\Delta t}{3.6\times 10^{6}}\\
& E^{\mathrm{pad}}_{k}  =\alpha^{\mathrm{pad}}\cdot U^{\mathrm{pad}}_{k} \cdot \Delta t\\
& E^{\mathrm{hum}}_{k}  =\alpha^{\mathrm{hum}}  \cdot \bar{F}^{\mathrm{hum}}   \cdot U^{\mathrm{hum}}_{k} \cdot \Delta t \\
& E^{\mathrm{deh}}_{k}=\alpha^{\mathrm{deh}}\cdot \bar{F}^{\mathrm{deh}} \cdot U^{\mathrm{deh}}_{k} \cdot \Delta t\\
& E^{\mathrm{dos}}_{k}  =\alpha^{\mathrm{dos}}  \cdot \bar{D}^{\mathrm{dos}}   \cdot U^{\mathrm{dos}}_{k} \cdot \Delta t
\end{align}

The MPC model is given in \eqref{MPC}.
\begin{subequations}\label{MPC}
\begin{flalign}
\min_{\boldsymbol{u}_{0},...,\boldsymbol{u}_{K-1}}
& \sum_{k=0}^{K-1} \gamma^{k} \left[
\begin{aligned}
&  E^{\mathrm{heat}}_{k}
 + E^{\mathrm{fan}}_{k}
 + E^{\mathrm{LED}}_{k}
 + E^{\mathrm{pad}}_{k}
 + E^{\mathrm{hum}}_{k}
 + E^{\mathrm{deh}}_{k}
 + E^{\mathrm{dos}}_{k} \\
& + \lambda^{\mathrm{T+}} S^{\mathrm{T+}}_{k} + \lambda^{\mathrm{T-}} S^{\mathrm{T-}}_{k}
  + \lambda^{\mathrm{H+}} S^{\mathrm{H+}}_{k} + \lambda^{\mathrm{H-}} S^{\mathrm{H-}}_{k}\\
& + \lambda^{\mathrm{C+}} S^{\mathrm{C+}}_{k} + \lambda^{\mathrm{C-}} S^{\mathrm{C-}}_{k}
  + \lambda^{\mathrm{L}}  S^{\mathrm{L}}_{k}
\end{aligned}
\right]                                                                                 \mspace{-400mu} &\notag\\
\text{s.t.}\quad
& \boldsymbol{x}_{0} = \boldsymbol{x}^{\mathrm{ini}},                                   \mspace{-400mu} & \\
& \boldsymbol{x}_{k+1} = \boldsymbol{M} \boldsymbol{x}_k + \boldsymbol{N} \boldsymbol{u}_k + \boldsymbol{O} \boldsymbol{d}_k + \boldsymbol{m},
\mspace{-400mu} & k = 0,...,K-1 \\
& \underline{T}^{\mathrm{in}}_{k} - S^{\mathrm{T-}}_{k} \le T^{\mathrm{in}}_{k}\le \bar{T}^{\mathrm{in}}_{k} + S^{\mathrm{T+}}_{k},
\mspace{-400mu} & k = 0,...,K-1 \\
& \underline{H}^{\mathrm{in}}_{k} - S^{\mathrm{H-}}_{k} \le H^{\mathrm{in}}_{k}\le \bar{H}^{\mathrm{in}}_{k} + S^{\mathrm{H+}}_{k},
\mspace{-400mu} & k = 0,...,K-1 \\
& \underline{C}^{\mathrm{in}}_{k} - S^{\mathrm{C-}}_{k} \le C^{\mathrm{in}}_{k}\le \bar{C}^{\mathrm{in}}_{k} + S^{\mathrm{C+}}_{k},
\mspace{-400mu} & k = 0,...,K-1 \\
& S^{\mathrm{L}}_{k} \ge L^{\star}_{k} - L_{k},
\mspace{-400mu} & k = 0,...,K-1 \\
& S^{\mathrm{T+}}_{k}, S^{\mathrm{T-}}_{k},  S^{\mathrm{H+}}_{k}, S^{\mathrm{H-}}_{k}, S^{\mathrm{C+}}_{k}, S^{\mathrm{C-}}_{k}, S^{\mathrm{L}}_{k} \geq 0
\mspace{-400mu} & k = 0,...,K-1 \\
& 0 \le U^{\mathrm{heat}}_{k}      \le Y^{\mathrm{heat}}_{k},       \mspace{-400mu} & k = 0,...,K-1 \\
& 0 \le U^{\mathrm{fan}}_{k}       \le Y^{\mathrm{fan}}_{k},        \mspace{-400mu} & k = 0,...,K-1 \\
& 0 \le U^{\mathrm{nat}}_{k}       \le Y^{\mathrm{nat}}_{k},        \mspace{-400mu} & k = 0,...,K-1 \\
& 0 \le U^{\mathrm{pad}}_{k}       \le Y^{\mathrm{pad}}_{k},        \mspace{-400mu} & k = 0,...,K-1 \\
& 0 \le U^{\mathrm{dos}}_{k}       \le Y^{\mathrm{dos}}_{k},        \mspace{-400mu} & k = 0,...,K-1 \\
& 0 \le U^{\mathrm{LED}}_{k}       \le Y^{\mathrm{LED}}_{k},        \mspace{-400mu} & k = 0,...,K-1 \\
& 0 \le U^{\mathrm{shad}}_{k}      \le Y^{\mathrm{shad}}_{k},       \mspace{-400mu} & k = 0,...,K-1 \\
& 0 \le U^{\mathrm{warm}}_{k}      \le Y^{\mathrm{warm}}_{k},       \mspace{-400mu} & k = 0,...,K-1 \\
& 0 \le U^{\mathrm{hum}}_{k}       \le Y^{\mathrm{hum}}_{k},        \mspace{-400mu} & k = 0,...,K-1 \\
& 0 \le U^{\mathrm{deh}}_{k}       \le Y^{\mathrm{deh}}_{k},        \mspace{-400mu} & k = 0,...,K-1 \\
& U^{\mathrm{fan}}_{k} \geq U^{\mathrm{pad}}_{k},
\mspace{-400mu} & k = 0,...,K-1 \\
& U^{\mathrm{fan}}_{k} \le Y^{\mathrm{fan}}_{k},\, U^{\mathrm{dos}}_{k} \le Y^{\mathrm{dos}}_{k},\, Y^{\mathrm{fan}}_{k} + Y^{\mathrm{dos}}_{k} \le 1
\mspace{-400mu} & k = 0,...,K-1 \\
& U^{\mathrm{hum}}_{k} \le Y^{\mathrm{hum}}_{k},\, U^{\mathrm{deh}}_{k} \le Y^{\mathrm{deh}}_{k},\, Y^{\mathrm{hum}}_{k} + Y^{\mathrm{deh}}_{k} \le 1
\mspace{-400mu} & k = 0,...,K-1 \\
& Y^{\mathrm{heat}}_{k},
  Y^{\mathrm{fan}}_{k},
  Y^{\mathrm{nat}}_{k},
  Y^{\mathrm{pad}}_{k},
  Y^{\mathrm{dos}}_{k},
  Y^{\mathrm{LED}}_{k},
  Y^{\mathrm{hum}}_{k},
  Y^{\mathrm{deh}}_{k},
  Y^{\mathrm{shad}}_{k},
  Y^{\mathrm{warm}}_{k} \in \{0,1\},
\mspace{-400mu} & \notag\\
&
\mspace{-400mu} & k = 0,...,K-1
\end{flalign}
\end{subequations}
where $\Delta \boldsymbol{x}_{k}$ is the state incremental term; $\boldsymbol{M}$, $\boldsymbol{N}$, and $\boldsymbol{O}$ are the discrete-time system matrices, and $\boldsymbol{m}$ is the constant offset term.
